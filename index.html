Settings

  • autoplay

    • true
    • false

  • speed

    • number(seconds*1000)

  • slidesToScroll

    • number

  • orientation

    • 'h' (horizontal)
    • 'v' (vertical)

  • marginsBtwnSlides

    • number

  • waResponsive

    • true
    • false

  • arrows

    • true
    • false

  • prevArrow

    • string(html element)

  • nextArrow

    • string(html element)

  • dots

    • true
    • false

  • swipe

    • true
    • false

  • animation

    • 'ltr'(lefToRight)
    • 'rtl'(rightToLeft)
    • 'dtu'(downToUp)
    • 'utd'(upToDown)
    • 'cssEase'
    • 'fromCenter'
    • 'carousel'
    • 'none'

... 
<head>
  ... 
  <link rel="stylesheet" href="path/waslider.css">
  ...
</head>

...
<div class = "slider"> 
  ...
</div>
...

<footer>
  ...
  <!--add jquery here-->
  <script src="path/wasldier.js"< >/script>
  <!--add your js file here-->
  ...
</footer>
...

            ...

            ...
